108void logToSD(const SensorData_t &d) {
109 String dir = // Genera nombre de carpeta basado en fecha actual.
110 fechaCarpeta();
111 if (!SD_MMC.exists(dir)) SD_MMC.mkdir(dir);
112 String file = dir + // Genera nombre de archivo basado en la hora actual.
113 horaFichero();
114 File f = SD_MMC.open(file, FILE_APPEND);
115 if (!f) {
116 Serial.println("❌ Error abriendo " + file);
117 return;
118 }
119 // Formato CSV: hora,temp,light,soil
120 f.printf("%s,%.2f,%.0f,%.0f\n",
121 timestamp().c_str(), d.temp, d.light, d.soil);
122 f.close();
123}

142// ——————————————————————————
143// TAREA FreeRTOS: procesa datos y envía comandos
144// ——————————————————————————
149void TaskProcess(void *pv) {
150 SensorData_t d;
151 ActuatorCmd_t cmd;
152 for (;;) {
153 if (xQueueReceive(sensorQueue, &d, portMAX_DELAY)) {
154 // 1) Log en SD
155 logToSD(d);
156
157 // 2) Alertas Telegram
158 if (d.temp > T_THRESH) bot.sendMessage(CHAT_ID, "🌡 Temp > " + String(d.temp), "");
159 if (d.light > L_THRESH) bot.sendMessage(CHAT_ID, "💡 Luz > " + String(d.light), "");
160 if (d.soil > S_THRESH) bot.sendMessage(CHAT_ID, "💦 Suelo > " + String(d.soil), "");
161
162 // 3) Decisión de actuadores
163 cmd.fanOn = (d.temp > T_THRESH);
164 cmd.pumpOn = (d.soil > S_THRESH);
165 cmd.rgbOn = (d.light > L_THRESH);
166 cmd.buzzerOn = (d.soil > S_THRESH);
167
168 // 4) Envío comando a Actuadores
169 esp_now_send(actuatorMac, (uint8_t*)&cmd, sizeof(cmd));
170 }
171 }
172}

197
198 // 4) Inicializar ESP-NOW
199 if (esp_now_init() != ESP_OK) {
200 Serial.println("❌ Error al iniciar ESP-NOW");
201 while (1) delay(1000);
202 }
203 esp_now_register_recv_cb(onSensorRecv);
204
205 // 5) Añadir peer del nodo Sensores
206 {
207 esp_now_peer_info_t peer = {};
208 memcpy(peer.peer_addr, sensorMac, 6);
209 peer.channel = 0;
210 peer.encrypt = false;
211 esp_now_add_peer(&peer);
212 }
213 // 6) Añadir peer del nodo Actuadores
214 {
215 esp_now_peer_info_t peer = {};
216 memcpy(peer.peer_addr, actuatorMac, 6);
217 peer.channel = 0;
218 peer.encrypt = false;
219 esp_now_add_peer(&peer);
220 }
221
222 // 7) Crear cola y lanzar tarea de procesamiento
223 sensorQueue = xQueueCreate(4, sizeof(SensorData_t));
224 xTaskCreatePinnedToCore(
225 TaskProcess, // función
226 "ProcSens", // nombre
227 8192, // stack
228 NULL, // parámetro
229 1, // prioridad
230 NULL, // handle
231 1 // core
232 );
233}
