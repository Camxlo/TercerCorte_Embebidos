43// ——————————————————————————
44// Callback: se llama al recibir un comando ESP-NOW

52void onCmdRecv(const esp_now_recv_info_t* info, const uint8_t* data, int len) {
53 if (len != sizeof(ActuatorCmd_t)) return;
54 ActuatorCmd_t cmd;
55 memcpy(&cmd, data, sizeof(cmd));
56
57 digitalWrite(PIN_FAN, cmd.fanOn ? HIGH : LOW);
58 digitalWrite(PIN_PUMP, cmd.pumpOn ? HIGH : LOW);
59 digitalWrite(PIN_RGB, cmd.rgbOn ? HIGH : LOW);
60 digitalWrite(PIN_BUZZER, cmd.buzzerOn ? HIGH : LOW);
61
62 Serial.printf("Cmd → Fan:%s Pump:%s RGB:%s Buzzer:%s\n",
63 cmd.fanOn ? "ON" : "OFF",
64 cmd.pumpOn ? "ON" : "OFF",
65 cmd.rgbOn ? "ON" : "OFF",
66 cmd.buzzerOn ? "ON" : "OFF");
67}

72void setup() {

74 Serial.begin(115200);
75

83 WiFi.mode(WIFI_STA);
84
85 // Inicializa ESP-NOW
86 if (esp_now_init() != ESP_OK) {
87 Serial.println("❌ Error al iniciar ESP-NOW");
88 while (true) { delay(1000); }
89 }
90
91 // Registra callback para recepción
92 esp_now_register_recv_cb(onCmdRecv);
93
94 // Añade el controlador como peer
95 esp_now_peer_info_t peer = {};
96 memcpy(peer.peer_addr, controllerMac, 6);
97 peer.channel = 0;
98 peer.encrypt = false;
99 if (esp_now_add_peer(&peer) != ESP_OK) {
100 Serial.println("⚠️ Error al agregar peer Controlador");
101 }
102}
